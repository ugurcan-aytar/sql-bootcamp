EXPLAIN ile Sorgu Analizi (Query Analysis)

EXPLAIN Komutlari (Commands)
-- Temel explain (sadece yurutme planini / execution plan gosterir)
EXPLAIN
SELECT * FROM employees WHERE department_id = 5;

-- Sorguyu gercekten calistir (gercek zamani gosterir)
EXPLAIN ANALYZE
SELECT * FROM employees WHERE department_id = 5;

-- Buffer'larla ayrintili (I/O bilgisi)
EXPLAIN (ANALYZE, BUFFERS, FORMAT TEXT)
SELECT * FROM employees WHERE salary > 50000;

-- JSON format (gorsellestirme araclari icin)
EXPLAIN (ANALYZE, FORMAT JSON)
SELECT * FROM employees;

Index Olusturma (Creating Indexes)

Index Sozdizimi (Syntax)
-- Temel B-tree index (varsayilan)
CREATE INDEX idx_employees_dept ON employees (department_id);

-- Unique index
CREATE UNIQUE INDEX idx_users_email ON users (email);

-- Coklu sutun index (Composite Index)
CREATE INDEX idx_orders_customer_date
ON orders (customer_id, order_date DESC);

-- Partial index / Kismi index (kosullu)
CREATE INDEX idx_orders_pending
ON orders (created_at)
WHERE status = 'pending';

-- Ifade index'i (Expression Index)
CREATE INDEX idx_users_email_lower
ON users (LOWER(email));

-- Eszamanli index / Concurrent index (kilit yok)
CREATE INDEX CONCURRENTLY idx_large_table
ON large_table (column);
Index Turleri (Index Types)
-- B-tree (varsayilan) - karsilastirmalar
CREATE INDEX idx_btree
ON table USING BTREE (column);

-- Hash - sadece esitlik
CREATE INDEX idx_hash
ON table USING HASH (column);

-- GIN - diziler, JSONB, tam metin
CREATE INDEX idx_gin
ON table USING GIN (jsonb_column);

-- GiST - geometrik, aralik tipleri
CREATE INDEX idx_gist
ON table USING GIST (location);

-- BRIN - buyuk sirali tablolar
CREATE INDEX idx_brin
ON table USING BRIN (created_at);

Index'leri Yonetme (Managing Indexes)
-- Tablodaki index'leri listele
SELECT indexname, indexdef
FROM pg_indexes
WHERE tablename = 'employees';

-- Index sil (Drop Index)
DROP INDEX idx_employees_dept;
DROP INDEX CONCURRENTLY idx_employees_dept;  -- Kilit yok

-- Index'i yeniden olustur (Rebuild Index)
REINDEX INDEX idx_employees_dept;
REINDEX TABLE employees;

-- Index boyutu (Index Size)
SELECT pg_size_pretty(pg_relation_size('idx_employees_dept'));

Sorgu Optimizasyonu (Query Optimization) Ipuclari

Bunu Yapin (Do This)
-- WHERE'de indexli sutunlari kullanin
WHERE indexed_column = value

-- Buyuk sonuclar icin LIMIT kullanin
SELECT * FROM logs
ORDER BY created_at DESC
LIMIT 100;

-- Alt sorgular (Subqueries) icin IN yerine EXISTS kullanin
WHERE EXISTS (SELECT 1 FROM ... WHERE ...)

-- Sadece gerekli sutunlari secin
SELECT id, name FROM users;

-- Kapsayici index'ler (Covering Indexes) kullanin
CREATE INDEX idx_covering
ON orders (customer_id) INCLUDE (total);
Bundan Kacinin (Avoid This)
-- Indexli sutunlarda fonksiyonlar
WHERE LOWER(email) = 'test@email.com'
-- Bunun yerine ifade index'i kullanin

-- Bastaki joker karakter (Leading Wildcard)
WHERE name LIKE '%john%'
-- B-tree index kullanilamaz

-- Farkli sutunlarda OR
WHERE col1 = x OR col2 = y
-- Bunun yerine UNION dusunun

-- JOIN'lerle SELECT *
SELECT * FROM a JOIN b JOIN c ...

-- != veya NOT IN (genellikle index kullanilamaz)
WHERE status != 'deleted'
-- Bunun yerine partial index kullanin

Istatistikler (Statistics) & Bakim (Maintenance)

Istatistikleri Guncelleme (Updating Statistics)
-- Sorgu planlayici (Query Planner) icin istatistikleri guncelle
ANALYZE employees;

-- Tum tablolar icin istatistikleri guncelle
ANALYZE;

-- VACUUM - olu tuple'lari (Dead Tuples) geri kazan
VACUUM employees;

-- VACUUM ANALYZE - her iki islem
VACUUM ANALYZE employees;

-- Tablo istatistiklerini goruntule (View Table Statistics)
SELECT relname, n_live_tup, n_dead_tup, last_vacuum, last_analyze
FROM pg_stat_user_tables;

Analitik Sorgular (Analytical Queries)

Pencere Fonksiyonlari (Window Functions) Ozeti
SELECT
    employee_name,
    department,
    salary,
    -- Siralama (Ranking)
    ROW_NUMBER() OVER (w) AS satir_no,
    RANK() OVER (w) AS siralama,
    DENSE_RANK() OVER (w) AS yogun_siralama,
    NTILE(4) OVER (w) AS ceyrek,

    -- Toplamalar (Aggregations)
    SUM(salary) OVER (PARTITION BY department) AS dept_toplam,
    AVG(salary) OVER () AS sirket_ort,

    -- Kumulatif toplamlar (Cumulative / Running Totals)
    SUM(salary) OVER (ORDER BY hire_date) AS kumulatif_toplam

FROM employees
WINDOW w AS (PARTITION BY department ORDER BY salary DESC);
LAG / LEAD
-- Onceki/sonraki satirlarla karsilastir (Compare with Previous/Next Rows)
SELECT
    order_date,
    amount,
    LAG(amount, 1) OVER (ORDER BY order_date)
        AS onceki_tutar,
    LEAD(amount, 1) OVER (ORDER BY order_date)
        AS sonraki_tutar,
    amount - LAG(amount) OVER (ORDER BY order_date)
        AS degisim
FROM daily_sales;
FIRST_VALUE / LAST_VALUE
SELECT
    name,
    salary,
    FIRST_VALUE(name) OVER (
        PARTITION BY department
        ORDER BY salary DESC
    ) AS en_yuksek_maasli,
    LAST_VALUE(name) OVER (
        PARTITION BY department
        ORDER BY salary DESC
        ROWS BETWEEN UNBOUNDED PRECEDING
        AND UNBOUNDED FOLLOWING
    ) AS en_dusuk_maasli
FROM employees;
Cerceve Cumleleri (Frame Clauses)
-- Hareketli ortalama / Moving Average (son 3 satir)
AVG(amount) OVER (
    ORDER BY date
    ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
)

-- Partisyondaki tum satirlar (All Rows in Partition)
SUM(amount) OVER (
    PARTITION BY category
    ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
)

-- Aralik tabanli / Range-based (pozisyona degil degere gore)
SUM(amount) OVER (
    ORDER BY date
    RANGE BETWEEN INTERVAL '7 days' PRECEDING AND CURRENT ROW
)

Gelismis Gruplama (Advanced Grouping)

GROUPING SETS, ROLLUP, CUBE
-- GROUPING SETS: belirli gruplama kombinasyonlari
SELECT region, product, SUM(sales)
FROM sales
GROUP BY GROUPING SETS (
    (region, product),  -- Bolge ve urune gore
    (region),           -- Sadece bolgeye gore
    (product),          -- Sadece urune gore
    ()                  -- Genel toplam
);

-- ROLLUP: hiyerarsik ara toplamlar (Hierarchical Subtotals)
SELECT year, quarter, month, SUM(sales)
FROM sales
GROUP BY ROLLUP (year, quarter, month);
-- Uretir: (year,quarter,month), (year,quarter), (year), ()

-- CUBE: tum olasi kombinasyonlar
SELECT region, product, SUM(sales)
FROM sales
GROUP BY CUBE (region, product);
-- Uretir: (region,product), (region), (product), ()

Yaygin SQL Kaliplari

Recursive CTE'ler (Hiyerarsiler)
WITH RECURSIVE tree AS (
    -- Capa: baslangic noktasi
    SELECT id, name, parent_id, 1 AS lvl
    FROM categories
    WHERE parent_id IS NULL

    UNION ALL

    -- Recursive: kendine birlestir
    SELECT c.id, c.name, c.parent_id, t.lvl + 1
    FROM categories c
    JOIN tree t ON c.parent_id = t.id
)
SELECT * FROM tree;
Grup Basina En Iyi N
-- ROW_NUMBER yontemi
WITH ranked AS (
    SELECT *,
        ROW_NUMBER() OVER (
            PARTITION BY category_id
            ORDER BY price DESC
        ) AS rn
    FROM products
)
SELECT * FROM ranked WHERE rn <= 3;

-- DISTINCT ON (En Iyi 1, PostgreSQL)
SELECT DISTINCT ON (customer_id)
    * FROM orders
ORDER BY customer_id, order_date DESC;
Bosluk ve Ada (Ardisik Gruplar)
-- Adalari bul (ardisik degerler)
WITH grouped AS (
    SELECT id,
        id - ROW_NUMBER() OVER (
            ORDER BY id
        ) AS grp
    FROM order_ids
)
SELECT
    MIN(id) AS baslangic_id,
    MAX(id) AS bitis_id
FROM grouped
GROUP BY grp;
Veri Pivotlama (Satirdan Sutuna)
-- FILTER yan tumcesiyle
SELECT
    kategori,
    SUM(tutar) FILTER (
        WHERE ay = 1
    ) AS oca,
    SUM(tutar) FILTER (
        WHERE ay = 2
    ) AS sub
FROM satislar
GROUP BY kategori;

-- CASE kullanarak
SUM(CASE WHEN ay=1 THEN tutar END)
Tarih Araligi Cakisma Tespiti
-- Iki aralik cakisir eger: baslangic1 <= bitis2 AND bitis1 >= baslangic2
SELECT * FROM rezervasyonlar a
JOIN rezervasyonlar b ON a.oda_id = b.oda_id AND a.id < b.id
WHERE a.baslangic <= b.bitis AND a.bitis >= b.baslangic;

-- PostgreSQL daterange tipi kullanarak
WHERE daterange(baslangic1, bitis1) && daterange(baslangic2, bitis2)  -- Cakisir
WHERE daterange(baslangic1, bitis1) @> '2024-06-15'::date          -- Tarihi icerir

Performans En Iyi Uygulamalari (Performance Best Practices)

Foreign key'leri (Yabanci Anahtarlar) indexleyin - JOIN'ler icin FK sutunlarini her zaman indexleyin
EXPLAIN ANALYZE kullanin - Tahmin etmeyin, olcun
Yavas sorgulari (Slow Queries) izleyin - pg_stat_statements'i etkinlestirin
Buyuk islemleri parcalayin (Batch Processing) - Parcalar halinde isleyin
Asiri indexleme yapmayin (Over-indexing) - Her index yazmalari yavaslatir
Seq Scan'lere dikkat edin - Buyuk tablolarda = problem
Sisme kontrolu yapin (Table Bloat) - VACUUM'u duzenly calistirin
Tahmin vs gercek (Estimated vs Actual) - Buyuk fark = guncel olmayan istatistikler

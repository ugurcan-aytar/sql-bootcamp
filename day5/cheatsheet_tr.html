Stored Procedure'lar & Fonksiyonlar

Fonksiyon Olusturma
-- Skaler donduren basit fonksiyon
CREATE OR REPLACE FUNCTION get_employee_count(dept_id INT)
RETURNS INTEGER AS $$
BEGIN
    RETURN (SELECT COUNT(*) FROM employees WHERE department_id = dept_id);
END;
$$ LANGUAGE plpgsql;

-- Kullanim
SELECT get_employee_count(5);

-- Tablo donduren fonksiyon
CREATE OR REPLACE FUNCTION get_employees_by_dept(dept_id INT)
RETURNS TABLE (id INT, name VARCHAR, salary NUMERIC) AS $$
BEGIN
    RETURN QUERY
    SELECT e.id, e.name, e.salary
    FROM employees e
    WHERE e.department_id = dept_id;
END;
$$ LANGUAGE plpgsql;

-- Kullanim
SELECT * FROM get_employees_by_dept(5);
Stored Procedure'lar (PostgreSQL 11+)
-- Procedure transaction kullanabilir
CREATE OR REPLACE PROCEDURE transfer_funds(
    sender_id INT,
    receiver_id INT,
    amount NUMERIC
)
LANGUAGE plpgsql AS $$
BEGIN
    -- Gonderenden dus
    UPDATE accounts SET balance = balance - amount
    WHERE id = sender_id;

    -- Aliciya ekle
    UPDATE accounts SET balance = balance + amount
    WHERE id = receiver_id;

    COMMIT;
END;
$$;

-- Procedure cagir
CALL transfer_funds(1, 2, 100.00);

Trigger'lar

Trigger Olusturma
-- Adim 1: Trigger fonksiyonu olustur
CREATE OR REPLACE FUNCTION update_modified_timestamp()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at := NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Adim 2: Trigger olustur
CREATE TRIGGER set_updated_at
    BEFORE UPDATE ON employees
    FOR EACH ROW
    EXECUTE FUNCTION update_modified_timestamp();
Trigger Zamanlama
BEFOREIslemden once, NEW'i degistirebilir
AFTERIslemden sonra, loglama icin
INSTEAD OFIslemi degistir (sadece view'lar)
Trigger Degiskenleri
NEWYeni satir (INSERT/UPDATE)
OLDEski satir (UPDATE/DELETE)
TG_OP'INSERT', 'UPDATE', 'DELETE'
TG_TABLE_NAMETrigger'i baslatan tablo
Denetim Trigger'i Ornegi
CREATE TABLE audit_log (
    id SERIAL PRIMARY KEY,
    table_name TEXT,
    operation TEXT,
    old_data JSONB,
    new_data JSONB,
    changed_at TIMESTAMPTZ DEFAULT NOW(),
    changed_by TEXT DEFAULT current_user
);

CREATE OR REPLACE FUNCTION audit_trigger()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO audit_log (table_name, operation, old_data, new_data)
    VALUES (
        TG_TABLE_NAME,
        TG_OP,
        CASE WHEN TG_OP = 'DELETE' THEN to_jsonb(OLD) ELSE NULL END,
        CASE WHEN TG_OP IN ('INSERT', 'UPDATE') THEN to_jsonb(NEW) ELSE NULL END
    );
    RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER audit_employees
    AFTER INSERT OR UPDATE OR DELETE ON employees
    FOR EACH ROW EXECUTE FUNCTION audit_trigger();

Normal View'lar
-- View olustur
CREATE VIEW active_employees AS
SELECT e.id, e.name, d.name AS dept
FROM employees e
JOIN departments d ON e.dept_id = d.id
WHERE e.status = 'active';

Materialized View'lar
-- Veriyi fiziksel olarak saklar
CREATE MATERIALIZED VIEW mv_sales_summary
AS
SELECT product_id,
       SUM(amount) AS total
FROM sales
GROUP BY product_id;

-- Guncellemek icin yenilenmeli
REFRESH MATERIALIZED VIEW mv_sales_summary;

Recursive Sorgular

Recursive CTE
-- Hiyerarsik veri (organizasyon semasi, kategoriler)
WITH RECURSIVE org_tree AS (
    -- Temel durum: ust duzey yoneticiler
    SELECT id, name, manager_id, 1 AS level
    FROM employees
    WHERE manager_id IS NULL

    UNION ALL

    -- Recursive durum: astlar
    SELECT e.id, e.name, e.manager_id, t.level + 1
    FROM employees e
    JOIN org_tree t ON e.manager_id = t.id
)
SELECT * FROM org_tree ORDER BY level, name;
Yol Izleme Ornegi
-- Kategoriler icin tam yol olustur
WITH RECURSIVE category_path AS (
    SELECT id, name, parent_id,
           name::TEXT AS path,
           ARRAY[id] AS path_ids
    FROM categories
    WHERE parent_id IS NULL

    UNION ALL

    SELECT c.id, c.name, c.parent_id,
           cp.path || ' > ' || c.name,
           cp.path_ids || c.id
    FROM categories c
    JOIN category_path cp ON c.parent_id = cp.id
)
SELECT id, name, path FROM category_path;

Yedekleme & Kurtarma

pg_dump - Mantiksal Yedekleme
# Tek veritabani yedekle (SQL formati)
pg_dump -h localhost -U postgres mydb > backup.sql

# Ozel format (sikistirilmis, paralel geri yukleme)
pg_dump -Fc -h localhost -U postgres mydb > backup.dump

# Dizin formati (paralel yedekleme)
pg_dump -Fd -j 4 -h localhost -U postgres mydb -f backup_dir/

# Sadece sema (veri yok)
pg_dump --schema-only mydb > schema.sql

# Sadece veri (sema yok)
pg_dump --data-only mydb > data.sql

# Belirli tablolar
pg_dump -t employees -t departments mydb > tables.sql

# Tablo haric tut
pg_dump --exclude-table=logs mydb > backup.sql
Geri Yukleme Komutlari
# SQL formati geri yukle
psql -h localhost -U postgres mydb < backup.sql

# Ozel/dizin formati geri yukle
pg_restore -h localhost -U postgres -d mydb backup.dump

# Paralel geri yukleme (daha hizli)
pg_restore -j 4 -d mydb backup.dump

# Veritabani olustur ve geri yukle
pg_restore -C -d postgres backup.dump

# Temizle (sil) sonra geri yukle
pg_restore --clean -d mydb backup.dump

# Belirli tabloyu geri yukle
pg_restore -t employees -d mydb backup.dump
pg_basebackup - Fiziksel Yedekleme
# Tam cluster yedegi
pg_basebackup -h localhost -U postgres -D /backup/base -Fp -Xs -P

# Sikistirilmis yedek
pg_basebackup -h localhost -U postgres -D /backup/base -Ft -z -P

# Secenekler:
# -Fp: Duz format (dizin)
# -Ft: Tar formati
# -Xs: Yedekleme sirasinda WAL akisi
# -z: Ciktiyi sikistir
# -P: Ilerlemeyi goster

Kullanici & Rol Yonetimi

Rol & Kullanici Olusturma
-- Rol olustur
CREATE ROLE analyst;

-- Kullanici olustur (LOGIN'li rol)
CREATE USER john WITH PASSWORD 'guvenli_sifre';

-- Seceneklerle rol olustur
CREATE ROLE admin WITH
    LOGIN
    SUPERUSER
    CREATEDB
    CREATEROLE
    PASSWORD 'admin_sifre';

-- Rolu degistir
ALTER ROLE john WITH PASSWORD 'yeni_sifre';
ALTER ROLE john WITH CREATEDB;

-- Rolu sil
DROP ROLE john;
Yetki Verme
-- Tabloya yetki ver
GRANT SELECT ON employees TO analyst;
GRANT SELECT, INSERT, UPDATE ON employees TO app_user;
GRANT ALL PRIVILEGES ON employees TO admin;

-- Semadaki tum tablolara yetki ver
GRANT SELECT ON ALL TABLES IN SCHEMA public TO analyst;

-- Gelecek tablolar icin varsayilan yetkiler
ALTER DEFAULT PRIVILEGES IN SCHEMA public
GRANT SELECT ON TABLES TO analyst;

-- Rol uyeligini ver
GRANT analyst TO john;

-- Yetkileri geri al
REVOKE INSERT ON employees FROM app_user;
REVOKE ALL ON employees FROM john;

Izleme & Tani

Faydali Sistem View'lari
-- Aktif baglantılar
SELECT * FROM pg_stat_activity;

-- Su anda calisan sorgular
SELECT pid, now() - pg_stat_activity.query_start AS sure, query
FROM pg_stat_activity
WHERE state != 'idle' AND query NOT LIKE '%pg_stat_activity%'
ORDER BY sure DESC;

-- Tablo boyutlari
SELECT relname, pg_size_pretty(pg_total_relation_size(relid))
FROM pg_catalog.pg_statio_user_tables
ORDER BY pg_total_relation_size(relid) DESC;

-- Index kullanimi
SELECT relname, indexrelname, idx_scan, idx_tup_read
FROM pg_stat_user_indexes
ORDER BY idx_scan DESC;

-- Kullanilmayan index'ler
SELECT indexrelname FROM pg_stat_user_indexes
WHERE idx_scan = 0;
Kilit Yonetimi
-- Kilitleri goruntule
SELECT locktype, relation::regclass, mode, granted, pid
FROM pg_locks
WHERE NOT granted;

-- Engelleyen sorgulari bul
SELECT
    blocked.pid AS blocked_pid,
    blocking.pid AS blocking_pid,
    blocked.query AS blocked_query,
    blocking.query AS blocking_query
FROM pg_stat_activity blocked
JOIN pg_stat_activity blocking ON blocking.pid = ANY(pg_blocking_pids(blocked.pid));

-- Sorguyu iptal et
SELECT pg_cancel_backend(pid);

-- Baglantiyi sonlandir (zorla)
SELECT pg_terminate_backend(pid);

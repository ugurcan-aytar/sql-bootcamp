Bölüm 5: Gelismis PostgreSQL Ozellikleri

Gelismis PostgreSQL Ozellikleri

Sakli Prosedurler ve Fonksiyonlar (Stored Procedures and Functions)

PostgreSQL, is mantigi dogrudan veritabaninda kapsulleyebilen sakli prosedurler ve fonksiyonlar olusturmak icin guclu yetenekler sunar. Bu yaklasim, gelistirilmis performans, kod yeniden kullanimi ve guvenlik dahil olmak uzere cesitli avantajlar saglar.

Fonksiyonlar ve Prosedurler (Functions and Procedures):

PostgreSQL'de iki tur rutin vardir:

  1. Fonksiyonlar (Functions): Bir deger dondurur ve SQL ifadelerinde kullanilabilir
  2. Prosedurler (Procedures): Deger dondurmez ancak islem (transaction) yapabilir (COMMIT/ROLLBACK)

Temel Fonksiyonlar Olusturma (Creating Basic Functions):

-- Skaler deger donduren basit fonksiyon
CREATE OR REPLACE FUNCTION get_customer_count()
RETURNS integer AS $$
BEGIN
    RETURN (SELECT COUNT(*) FROM customer);
END;
$$ LANGUAGE plpgsql;

-- Kullanim
SELECT get_customer_count();

-- Parametreli fonksiyon
CREATE OR REPLACE FUNCTION get_orders_by_customer(customer_id_param integer)
RETURNS integer AS $$
BEGIN
    RETURN (SELECT COUNT(*) FROM orders WHERE customer_id = customer_id_param);
END;
$$ LANGUAGE plpgsql;

-- Kullanim
SELECT get_orders_by_customer(123);

Tablo Donduren Fonksiyonlar Olusturma (Creating Table-Returning Functions):

-- Tablo donduren fonksiyon
CREATE OR REPLACE FUNCTION get_top_customers(n integer)
RETURNS TABLE (
    customer_id integer,
    customer_name text,
    total_orders integer,
    total_spent numeric
) AS $$
BEGIN
    RETURN QUERY
    SELECT
        c.id,
        c.name,
        COUNT(o.id),
        SUM(o.amount)
    FROM
        customer c
    JOIN
        orders o ON c.id = o.customer_id
    GROUP BY
        c.id, c.name
    ORDER BY
        SUM(o.amount) DESC
    LIMIT n;
END;
$$ LANGUAGE plpgsql;

-- Kullanim
SELECT * FROM get_top_customers(10);

Sakli Prosedurler Olusturma (Creating Stored Procedures):

-- Basit prosedur
CREATE OR REPLACE PROCEDURE update_product_prices(category_id_param integer, increase_percent numeric)
LANGUAGE plpgsql
AS $$
BEGIN
    UPDATE product
    SET price = price * (1 + increase_percent / 100)
    WHERE category_id = category_id_param;
    
    COMMIT;
END;
$$;

-- Kullanim
CALL update_product_prices(5, 10);  -- Kategori 5'teki fiyatlari %10 artir

-- Islem kontrollu prosedur
CREATE OR REPLACE PROCEDURE transfer_funds(
    sender_id integer,
    receiver_id integer,
    amount numeric
)
LANGUAGE plpgsql
AS $$
DECLARE
    sender_balance numeric;
BEGIN
    -- Islem baslat
    BEGIN
        -- Gonderen bakiyesini kontrol et
        SELECT balance INTO sender_balance
        FROM account
        WHERE account_id = sender_id
        FOR UPDATE;  -- Satiri kilitle
        
        IF sender_balance < amount THEN
            RAISE EXCEPTION 'Yetersiz bakiye';
        END IF;
        
        -- Gonderen hesabini guncelle
        UPDATE account
        SET balance = balance - amount
        WHERE account_id = sender_id;
        
        -- Alici hesabini guncelle
        UPDATE account
        SET balance = balance + amount
        WHERE account_id = receiver_id;
        
        -- Islemi onayla
        COMMIT;
    EXCEPTION
        WHEN OTHERS THEN
            -- Hata durumunda islemi geri al
            ROLLBACK;
            RAISE;
    END;
END;
$$;

-- Kullanim
CALL transfer_funds(101, 202, 500.00);

PL/pgSQL Dil Ozellikleri (PL/pgSQL Language Features):

PL/pgSQL, SQL'i kontrol yapilari ve daha karmasik islemlerle genisleten PostgreSQL'in prosedural dilidir.

  1. Degiskenler ve Sabitler (Variables and Constants):
CREATE OR REPLACE FUNCTION calculate_order_total(order_id_param integer)
RETURNS numeric AS $$
DECLARE
    total_amount numeric := 0;
    tax_rate constant numeric := 0.08;
    shipping_fee numeric;
    customer_type text;
BEGIN
    -- Musteri turunu al
    SELECT c.customer_type INTO customer_type
    FROM orders o
    JOIN customer c ON o.customer_id = c.id
    WHERE o.id = order_id_param;
    
    -- Musteri turune gore kargo ucretini belirle
    IF customer_type = 'premium' THEN
        shipping_fee := 0;
    ELSE
        shipping_fee := 10;
    END IF;
    
    -- Siparis toplamini hesapla
    SELECT SUM(quantity * price) INTO total_amount
    FROM order_item
    WHERE order_id = order_id_param;
    
    -- Vergi ve kargoyu ekle
    total_amount := total_amount * (1 + tax_rate) + shipping_fee;
    
    RETURN total_amount;
END;
$$ LANGUAGE plpgsql;
  1. Kontrol Yapilari (Control Structures):
-- IF ifadesi
CREATE OR REPLACE FUNCTION get_price_category(price numeric)
RETURNS text AS $$
BEGIN
    IF price < 50 THEN
        RETURN 'Butce';
    ELSIF price < 200 THEN
        RETURN 'Orta-segment';
    ELSE
        RETURN 'Premium';
    END IF;
END;
$$ LANGUAGE plpgsql;

-- CASE ifadesi
CREATE OR REPLACE FUNCTION get_shipping_method(weight numeric)
RETURNS text AS $$
BEGIN
    RETURN CASE
        WHEN weight < 1 THEN 'Standart Posta'
        WHEN weight < 5 THEN 'Ekspres Posta'
        WHEN weight < 20 THEN 'Kara Kargo'
        ELSE 'Navlun'
    END;
END;
$$ LANGUAGE plpgsql;

-- LOOP
CREATE OR REPLACE FUNCTION sum_integers(n integer)
RETURNS integer AS $$
DECLARE
    i integer := 0;
    total integer := 0;
BEGIN
    LOOP
        i := i + 1;
        total := total + i;
        EXIT WHEN i >= n;
    END LOOP;
    
    RETURN total;
END;
$$ LANGUAGE plpgsql;

-- WHILE LOOP
CREATE OR REPLACE FUNCTION factorial(n integer)
RETURNS integer AS $$
DECLARE
    result integer := 1;
BEGIN
    WHILE n > 0 LOOP
        result := result * n;
        n := n - 1;
    END LOOP;
    
    RETURN result;
END;
$$ LANGUAGE plpgsql;

-- FOR LOOP (tamsayi araligi)
CREATE OR REPLACE FUNCTION sum_even_numbers(n integer)
RETURNS integer AS $$
DECLARE
    i integer;
    total integer := 0;
BEGIN
    FOR i IN 1..n LOOP
        IF i % 2 = 0 THEN
            total := total + i;
        END IF;
    END LOOP;
    
    RETURN total;
END;
$$ LANGUAGE plpgsql;

-- FOR LOOP (sorgu sonuclari)
CREATE OR REPLACE FUNCTION update_inventory_levels()
RETURNS void AS $$
DECLARE
    product_record record;
BEGIN
    FOR product_record IN
        SELECT product_id, name
        FROM product
        WHERE stock_level < reorder_level
    LOOP
        RAISE NOTICE 'Urun % icin dusuk stok: %', 
                     product_record.product_id, 
                     product_record.name;
        
        -- Yeniden siparis bayragini guncelle
        UPDATE product
        SET reorder_flag = true
        WHERE product_id = product_record.product_id;
    END LOOP;
END;
$$ LANGUAGE plpgsql;
  1. Hata Yonetimi (Error Handling):
CREATE OR REPLACE FUNCTION divide_safely(a numeric, b numeric)
RETURNS numeric AS $$
BEGIN
    RETURN a / b;
EXCEPTION
    WHEN division_by_zero THEN
        RAISE NOTICE 'Sifira bolme tespit edildi';
        RETURN 0;
    WHEN others THEN
        RAISE NOTICE 'Diger hata: %', SQLERRM;
        RETURN NULL;
END;
$$ LANGUAGE plpgsql;

-- Ozel istisnalar
CREATE OR REPLACE FUNCTION withdraw_money(account_id_param integer, amount numeric)
RETURNS void AS $$
DECLARE
    current_balance numeric;
    insufficient_funds EXCEPTION;
BEGIN
    -- Mevcut bakiyeyi al
    SELECT balance INTO current_balance
    FROM account
    WHERE account_id = account_id_param;
    
    -- Yeterli bakiye var mi kontrol et
    IF current_balance < amount THEN
        RAISE insufficient_funds;
    END IF;
    
    -- Bakiyeyi guncelle
    UPDATE account
    SET balance = balance - amount
    WHERE account_id = account_id_param;
    
EXCEPTION
    WHEN insufficient_funds THEN
        RAISE EXCEPTION 'Hesap % yetersiz bakiyeye sahip (bakiye: %)', 
                        account_id_param, current_balance;
    WHEN no_data_found THEN
        RAISE EXCEPTION 'Hesap % bulunamadi', account_id_param;
END;
$$ LANGUAGE plpgsql;

Fonksiyon Guvenligi ve Izinler:

Fonksiyonlar farkli guvenlik baglamlarinda calistirilabiir:

-- Fonksiyon cagiran kisi yetkileriyle calisir
CREATE OR REPLACE FUNCTION get_my_orders()
RETURNS TABLE (order_id integer, order_date date, amount numeric)
AS $$
BEGIN
    RETURN QUERY
    SELECT id, order_date, amount
    FROM orders
    WHERE customer_id = current_user_id();
END;
$$ LANGUAGE plpgsql;

-- Fonksiyon tanimlayan kisi yetkileriyle calisir
CREATE OR REPLACE FUNCTION get_sensitive_data()
RETURNS TABLE (customer_id integer, credit_score integer)
SECURITY DEFINER
AS $$
BEGIN
    RETURN QUERY
    SELECT id, credit_score
    FROM customer_credit_data;
END;
$$ LANGUAGE plpgsql;

-- Calistirma izni ver
GRANT EXECUTE ON FUNCTION get_sensitive_data() TO analyst_role;

Pratik Ornekler:

  1. Denetim Izi Fonksiyonu:
-- Denetim tablosu olustur
CREATE TABLE audit_log (
    log_id serial PRIMARY KEY,
    table_name text NOT NULL,
    operation text NOT NULL,
    record_id integer NOT NULL,
    changed_by text NOT NULL,
    changed_at timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
    old_data jsonb,
    new_data jsonb
);

-- Denetim fonksiyonu olustur
CREATE OR REPLACE FUNCTION audit_trigger_function()
RETURNS trigger AS $$
BEGIN
    IF (TG_OP = 'DELETE') THEN
        INSERT INTO audit_log (
            table_name,
            operation,
            record_id,
            changed_by,
            old_data
        )
        VALUES (
            TG_TABLE_NAME,
            TG_OP,
            OLD.id,
            current_user,
            to_jsonb(OLD)
        );
        RETURN OLD;
    ELSIF (TG_OP = 'UPDATE') THEN
        INSERT INTO audit_log (
            table_name,
            operation,
            record_id,
            changed_by,
            old_data,
            new_data
        )
        VALUES (
            TG_TABLE_NAME,
            TG_OP,
            NEW.id,
            current_user,
            to_jsonb(OLD),
            to_jsonb(NEW)
        );
        RETURN NEW;
    ELSIF (TG_OP = 'INSERT') THEN
        INSERT INTO audit_log (
            table_name,
            operation,
            record_id,
            changed_by,
            new_data
        )
        VALUES (
            TG_TABLE_NAME,
            TG_OP,
            NEW.id,
            current_user,
            to_jsonb(NEW)
        );
        RETURN NEW;
    END IF;
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

-- Tetikleyiciyi tabloya uygula
CREATE TRIGGER product_audit
AFTER INSERT OR UPDATE OR DELETE ON product
FOR EACH ROW EXECUTE FUNCTION audit_trigger_function();

Tetikleyiciler ve Olay Tabanli Programlama (Triggers and Event-Based Programming)

Tetikleyiciler, bir tabloda belirli olaylar meydana geldiginde otomatik olarak bir fonksiyon calistirmaniza olanak tanir. Is kurallari uygulamak, turetilmis verileri korumak ve denetim izleri olusturmak icin kullanislidirlar.

Tetikleyici Turleri (Trigger Types):

  1. Satir duzeyi tetikleyiciler (Row-level triggers): Etkilenen her satir icin bir kez calisir
  2. Ifade duzeyi tetikleyiciler (Statement-level triggers): SQL ifadesi basina bir kez calisir
  3. Oncesi tetikleyiciler (BEFORE triggers): Islemden once calisir
  4. Sonrasi tetikleyiciler (AFTER triggers): Islemden sonra calisir
  5. Yerine tetikleyiciler (INSTEAD OF triggers): Islemi degistirir (gorunumler/views icin)

Temel Tetikleyiciler Olusturma (Creating Basic Triggers):

-- Tetikleyici fonksiyonu olustur
CREATE OR REPLACE FUNCTION update_modified_column()
RETURNS trigger AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Tetikleyiciyi tabloya uygula
CREATE TRIGGER set_updated_at
BEFORE UPDATE ON product
FOR EACH ROW
EXECUTE FUNCTION update_modified_column();

Tetikleyici Zamanlama ve Olaylar (Trigger Timing and Events):

-- INSERT oncesi tetikleyici
CREATE TRIGGER check_product_price
BEFORE INSERT ON product
FOR EACH ROW
EXECUTE FUNCTION validate_product_price();

-- UPDATE sonrasi tetikleyici
CREATE TRIGGER notify_price_change
AFTER UPDATE OF price ON product
FOR EACH ROW
WHEN (OLD.price IS DISTINCT FROM NEW.price)
EXECUTE FUNCTION log_price_change();

-- Ifade duzeyi tetikleyici
CREATE TRIGGER cleanup_after_bulk_update
AFTER UPDATE ON inventory
FOR EACH STATEMENT
EXECUTE FUNCTION refresh_inventory_summary();

-- Bir tetikleyicide birden fazla olay
CREATE TRIGGER maintain_audit_log
AFTER INSERT OR UPDATE OR DELETE ON customer
FOR EACH ROW
EXECUTE FUNCTION audit_customer_changes();

Tetikleyici Verilerine Erisim:

Tetikleyici fonksiyonu icinde ozel degiskenlere erisebilirsiniz:

  • NEW: INSERT/UPDATE islemleri icin yeni satir
  • OLD: UPDATE/DELETE islemleri icin eski satir
  • TG_OP: Islem ('INSERT', 'UPDATE', 'DELETE')
  • TG_TABLE_NAME: Tablo adi
  • TG_WHEN: Zamanlama ('BEFORE', 'AFTER', 'INSTEAD OF')
  • TG_LEVEL: Seviye ('ROW', 'STATEMENT')

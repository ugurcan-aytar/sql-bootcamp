Sakli Prosedurler ve Fonksiyonlar

PostgreSQL, is mantigi dogrudan veritabaninda kapsulleyebilen sakli prosedurler ve fonksiyonlar olusturmak icin guclu yetenekler sunar. Bu yaklasim, gelismis performans, kod yeniden kullanimi ve guvenlik dahil olmak uzere cesitli avantajlar saglar.

Fonksiyonlar ve Prosedurler:

PostgreSQL'de iki tur rutin vardir:

  1. Fonksiyonlar: Bir deger dondurur ve SQL ifadelerinde kullanilabilir
  2. Prosedurler: Bir deger dondurmez ancak islemler gerceklestirebilir (COMMIT/ROLLBACK)

Temel Fonksiyonlar Olusturma:

-- Simple function that returns a scalar value
CREATE OR REPLACE FUNCTION get_customer_count()
RETURNS integer AS $$
BEGIN
    RETURN (SELECT COUNT(*) FROM customer);
END;
$$ LANGUAGE plpgsql;

-- Kullanim
SELECT get_customer_count();

-- Function with parameters
CREATE OR REPLACE FUNCTION get_orders_by_customer(customer_id_param integer)
RETURNS integer AS $$
BEGIN
    RETURN (SELECT COUNT(*) FROM orders WHERE customer_id = customer_id_param);
END;
$$ LANGUAGE plpgsql;

-- Kullanim
SELECT get_orders_by_customer(123);

Tablo Donduren Fonksiyonlar Olusturma:

-- Function that returns a table
CREATE OR REPLACE FUNCTION get_top_customers(n integer)
RETURNS TABLE (
    customer_id integer,
    customer_name text,
    total_orders integer,
    total_spent numeric
) AS $$
BEGIN
    RETURN QUERY
    SELECT
        c.id,
        c.name,
        COUNT(o.id),
        SUM(o.amount)
    FROM
        customer c
    JOIN
        orders o ON c.id = o.customer_id
    GROUP BY
        c.id, c.name
    ORDER BY
        SUM(o.amount) DESC
    LIMIT n;
END;
$$ LANGUAGE plpgsql;

-- Kullanim
SELECT * FROM get_top_customers(10);

Sakli Prosedurler Olusturma:

-- Simple procedure
CREATE OR REPLACE PROCEDURE update_product_prices(category_id_param integer, increase_percent numeric)
LANGUAGE plpgsql
AS $$
BEGIN
    UPDATE product
    SET price = price * (1 + increase_percent / 100)
    WHERE category_id = category_id_param;
    
    COMMIT;
END;
$$;

-- Kullanim
CALL update_product_prices(5, 10);  -- Kategori 5'teki fiyatlari %10 artir

-- Procedure with transaction control
CREATE OR REPLACE PROCEDURE transfer_funds(
    sender_id integer,
    receiver_id integer,
    amount numeric
)
LANGUAGE plpgsql
AS $$
DECLARE
    sender_balance numeric;
BEGIN
    -- Start a transaction
    BEGIN
        -- Check sender balance
        SELECT balance INTO sender_balance
        FROM account
        WHERE account_id = sender_id
        FOR UPDATE;  -- Lock the row
        
        IF sender_balance < amount THEN
            RAISE EXCEPTION 'Insufficient funds';
        END IF;
        
        -- Update sender account
        UPDATE account
        SET balance = balance - amount
        WHERE account_id = sender_id;
        
        -- Update receiver account
        UPDATE account
        SET balance = balance + amount
        WHERE account_id = receiver_id;
        
        -- Commit the transaction
        COMMIT;
    EXCEPTION
        WHEN OTHERS THEN
            -- Rollback the transaction on error
            ROLLBACK;
            RAISE;
    END;
END;
$$;

-- Kullanim
CALL transfer_funds(101, 202, 500.00);

PL/pgSQL Dil Ozellikleri:

PL/pgSQL, SQL'i kontrol yapilari ve daha karmasik islemlerle genisleten PostgreSQL'in prosedural dilidir.

  1. Degiskenler ve Sabitler:
CREATE OR REPLACE FUNCTION calculate_order_total(order_id_param integer)
RETURNS numeric AS $$
DECLARE
    total_amount numeric := 0;
    tax_rate constant numeric := 0.08;
    shipping_fee numeric;
    customer_type text;
BEGIN
    -- Get the customer type
    SELECT c.customer_type INTO customer_type
    FROM orders o
    JOIN customer c ON o.customer_id = c.id
    WHERE o.id = order_id_param;
    
    -- Set shipping fee based on customer type
    IF customer_type = 'premium' THEN
        shipping_fee := 0;
    ELSE
        shipping_fee := 10;
    END IF;
    
    -- Calculate order total
    SELECT SUM(quantity * price) INTO total_amount
    FROM order_item
    WHERE order_id = order_id_param;
    
    -- Add tax and shipping
    total_amount := total_amount * (1 + tax_rate) + shipping_fee;
    
    RETURN total_amount;
END;
$$ LANGUAGE plpgsql;
  1. Kontrol Yapilari:
-- IF statement
CREATE OR REPLACE FUNCTION get_price_category(price numeric)
RETURNS text AS $$
BEGIN
    IF price < 50 THEN
        RETURN 'Budget';
    ELSIF price < 200 THEN
        RETURN 'Mid-range';
    ELSE
        RETURN 'Premium';
    END IF;
END;
$$ LANGUAGE plpgsql;

-- CASE statement
CREATE OR REPLACE FUNCTION get_shipping_method(weight numeric)
RETURNS text AS $$
BEGIN
    RETURN CASE
        WHEN weight < 1 THEN 'Standard Mail'
        WHEN weight < 5 THEN 'Express Mail'
        WHEN weight < 20 THEN 'Ground Shipping'
        ELSE 'Freight'
    END;
END;
$$ LANGUAGE plpgsql;

-- LOOP
CREATE OR REPLACE FUNCTION sum_integers(n integer)
RETURNS integer AS $$
DECLARE
    i integer := 0;
    total integer := 0;
BEGIN
    LOOP
        i := i + 1;
        total := total + i;
        EXIT WHEN i >= n;
    END LOOP;
    
    RETURN total;
END;
$$ LANGUAGE plpgsql;

-- WHILE LOOP
CREATE OR REPLACE FUNCTION factorial(n integer)
RETURNS integer AS $$
DECLARE
    result integer := 1;
BEGIN
    WHILE n > 0 LOOP
        result := result * n;
        n := n - 1;
    END LOOP;
    
    RETURN result;
END;
$$ LANGUAGE plpgsql;

-- FOR LOOP (integer range)
CREATE OR REPLACE FUNCTION sum_even_numbers(n integer)
RETURNS integer AS $$
DECLARE
    i integer;
    total integer := 0;
BEGIN
    FOR i IN 1..n LOOP
        IF i % 2 = 0 THEN
            total := total + i;
        END IF;
    END LOOP;
    
    RETURN total;
END;
$$ LANGUAGE plpgsql;

-- FOR LOOP (query results)
CREATE OR REPLACE FUNCTION update_inventory_levels()
RETURNS void AS $$
DECLARE
    product_record record;
BEGIN
    FOR product_record IN
        SELECT product_id, name
        FROM product
        WHERE stock_level < reorder_level
    LOOP
        RAISE NOTICE 'Low stock for product %: %', 
                     product_record.product_id, 
                     product_record.name;
        
        -- Update the reorder flag
        UPDATE product
        SET reorder_flag = true
        WHERE product_id = product_record.product_id;
    END LOOP;
END;
$$ LANGUAGE plpgsql;
  1. Hata Yonetimi:
CREATE OR REPLACE FUNCTION divide_safely(a numeric, b numeric)
RETURNS numeric AS $$
BEGIN
    RETURN a / b;
EXCEPTION
    WHEN division_by_zero THEN
        RAISE NOTICE 'Division by zero detected';
        RETURN 0;
    WHEN others THEN
        RAISE NOTICE 'Other error: %', SQLERRM;
        RETURN NULL;
END;
$$ LANGUAGE plpgsql;

-- Custom exceptions
CREATE OR REPLACE FUNCTION withdraw_money(account_id_param integer, amount numeric)
RETURNS void AS $$
DECLARE
    current_balance numeric;
    insufficient_funds EXCEPTION;
BEGIN
    -- Get current balance
    SELECT balance INTO current_balance
    FROM account
    WHERE account_id = account_id_param;
    
    -- Check if enough funds
    IF current_balance < amount THEN
        RAISE insufficient_funds;
    END IF;
    
    -- Update balance
    UPDATE account
    SET balance = balance - amount
    WHERE account_id = account_id_param;
    
EXCEPTION
    WHEN insufficient_funds THEN
        RAISE EXCEPTION 'Account % has insufficient funds (balance: %)', 
                        account_id_param, current_balance;
    WHEN no_data_found THEN
        RAISE EXCEPTION 'Account % not found', account_id_param;
END;
$$ LANGUAGE plpgsql;

Fonksiyon Guvenligi ve Izinler:

Fonksiyonlar farkli guvenlik baglamlarinda calistirabilir:

-- Function executes with the privileges of the caller
CREATE OR REPLACE FUNCTION get_my_orders()
RETURNS TABLE (order_id integer, order_date date, amount numeric)
AS $$
BEGIN
    RETURN QUERY
    SELECT id, order_date, amount
    FROM orders
    WHERE customer_id = current_user_id();
END;
$$ LANGUAGE plpgsql;

-- Function executes with the privileges of the definer
CREATE OR REPLACE FUNCTION get_sensitive_data()
RETURNS TABLE (customer_id integer, credit_score integer)
SECURITY DEFINER
AS $$
BEGIN
    RETURN QUERY
    SELECT id, credit_score
    FROM customer_credit_data;
END;
$$ LANGUAGE plpgsql;

-- Grant execute permission
GRANT EXECUTE ON FUNCTION get_sensitive_data() TO analyst_role;

Tetikleyiciler ve Olay Tabanli Programlama

Tetikleyiciler, bir tabloda belirli olaylar gerceklestiginde otomatik olarak bir fonksiyonun calistirilmasini saglar. Is kurallari uygulamak, turetilmis verileri korumak ve denetim izleri olusturmak icin kullanislidirlar.

Tetikleyici Turleri:

  1. Satir duzeyinde tetikleyiciler: Etkilenen her satir icin bir kez calisir
  2. Ifade duzeyinde tetikleyiciler: Her SQL ifadesi icin bir kez calisir
  3. Once tetikleyicileri: Islemden once calisir
  4. Sonra tetikleyicileri: Islemden sonra calisir
  5. Yerine tetikleyicileri: Islemi degistirir (gorunumler icin)

Temel Tetikleyiciler Olusturma:

-- Create a trigger function
CREATE OR REPLACE FUNCTION update_modified_column()
RETURNS trigger AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Apply the trigger to a table
CREATE TRIGGER set_updated_at
BEFORE UPDATE ON product
FOR EACH ROW
EXECUTE FUNCTION update_modified_column();

Tetikleyici Zamanlama ve Olaylari:

Bkz. orijinal dosyadaki tetikleyici ornekleri (BEFORE INSERT, AFTER UPDATE, vb.)

Tetikleyici Verilerine Erisim:

Bir tetikleyici fonksiyonu icinde ozel degiskenlere erisebilirsiniz:

  • NEW: INSERT/UPDATE islemleri icin yeni satir
  • OLD: UPDATE/DELETE islemleri icin eski satir
  • TG_OP: Islem ('INSERT', 'UPDATE', 'DELETE')
  • TG_TABLE_NAME: Tablo adi
  • TG_WHEN: Zamanlama ('BEFORE', 'AFTER', 'INSTEAD OF')
  • TG_LEVEL: Seviye ('ROW', 'STATEMENT')

Tetikleyicileri Devre Disi Birakma ve Etkinlestirme:

-- Disable a specific trigger
ALTER TABLE product DISABLE TRIGGER set_updated_at;

-- Disable all triggers on a table
ALTER TABLE product DISABLE TRIGGER ALL;

-- Enable a specific trigger
ALTER TABLE product ENABLE TRIGGER set_updated_at;

-- Enable all triggers on a table
ALTER TABLE product ENABLE TRIGGER ALL;

Pratik Tetikleyici Ornekleri:

  1. Denormalize Ozet Tablosu Koruma
  2. Is Kurallarini Uygulama
  3. Yumusak Silme Uygulama

Gelismis Indeksleme ve Sorgu Optimizasyonu

Daha onceki indeksleme tartismamizi temel alarak, bazi gelismis indeksleme teknikleri ve sorgu optimizasyonu stratejilerini inceleyelim.

Karmasik Kosullarla Kismi Indeksler:

-- Partial index for active orders
CREATE INDEX idx_active_orders ON orders(order_date)
WHERE status IN ('pending', 'processing');

-- Partial index for high-value products
CREATE INDEX idx_high_value_products ON product(name, category_id)
WHERE price > 1000;

Karmasik Sorgular icin Fonksiyonel Indeksler:

-- Index for case-insensitive search
CREATE INDEX idx_customer_name_lower ON customer(LOWER(name));

-- Index for date extraction
CREATE INDEX idx_order_year_month ON orders(
    EXTRACT(YEAR FROM order_date),
    EXTRACT(MONTH FROM order_date)
);

-- Index for JSON/JSONB paths
CREATE INDEX idx_product_attributes_material ON product((attributes->>'material'));

Performans icin Kapsayan Indeksler:

-- Include additional columns in the index
CREATE INDEX idx_order_customer_date_include_status ON orders(customer_id, order_date)
INCLUDE (status, amount);

Materyalize Gorunumlerle Optimizasyon:

-- Create a materialized view for expensive queries
CREATE MATERIALIZED VIEW product_sales_summary AS
SELECT
    p.product_id,
    p.name AS product_name,
    c.category_name,
    COUNT(DISTINCT oi.order_id) AS order_count,
    SUM(oi.quantity) AS units_sold,
    SUM(oi.quantity * oi.price) AS revenue
FROM
    product p
JOIN
    category c ON p.category_id = c.category_id
LEFT JOIN
    order_item oi ON p.product_id = oi.product_id
GROUP BY
    p.product_id, p.name, c.category_name;

-- Refresh the materialized view
REFRESH MATERIALIZED VIEW product_sales_summary;

Sorgu Optimizasyon Teknikleri:

  1. Alt Sorgulari Birlestirme Olarak Yeniden Yazma
  2. Karmasik Sorgular icin CTE Kullanma
  3. IN Cumlellerini Dizilerle Optimize Etme
  4. Satir Satir Isleme icin LATERAL Birlesimler Kullanma
  5. Pencere Fonksiyonlariyla Toplamalari Optimize Etme

Gelismis Performans Ayarlama:

  1. Tablo Bolumlendirme
  2. Hazirlanmis Ifadelerle Optimizasyon
  3. Uygulama Duzeyinde Kilitleme icin Advisory Locks Kullanma

JSON ve JSONB Verileriyle Calisma

PostgreSQL, yari yapilandirilmis verileri verimli bir sekilde depolamanizi ve sorgulamanizi saglayan guclu JSON destegi sunar.

JSON ve JSONB:

PostgreSQL iki JSON veri turu sunar:

  1. JSON: Bosluklar dahil JSON girisinin tam metnini saklar
  2. JSONB: JSON'u ikili formatta saklar, isleme icin daha verimlidir ancak bosluk veya anahtar sirasini korumaz
-- Create tables with JSON and JSONB columns
CREATE TABLE product_json (
    product_id serial PRIMARY KEY,
    name text NOT NULL,
    attributes JSON
);

CREATE TABLE product_jsonb (
    product_id serial PRIMARY KEY,
    name text NOT NULL,
    attributes JSONB
);

JSON Verilerini Sorgulama:

  1. JSON Elemanlarina Erisim:
-- Using -> operator (returns JSON)
SELECT attributes->'color' FROM product_json;

-- Using ->> operator (returns text)
SELECT attributes->>'color' FROM product_json;

-- Accessing nested elements
SELECT attributes->'features'->0 FROM product_json;
  1. JSON ile Filtreleme:
-- Filter by exact value
SELECT * FROM product_jsonb
WHERE attributes->>'color' = 'black';

-- JSONB containment operator (@>)
SELECT * FROM product_jsonb
WHERE attributes @> '{"color": "black", "storage": "128GB"}';
  1. JSON Verilerini Indeksleme:
-- GIN index for JSONB (supports @>, ?, ?|, ?& operators)
CREATE INDEX idx_product_attributes ON product_jsonb USING GIN (attributes);

-- B-tree index for specific JSON values
CREATE INDEX idx_product_color ON product_jsonb ((attributes->>'color'));

JSON Verilerini Degistirme:

-- Update a specific field
UPDATE product_jsonb
SET attributes = jsonb_set(attributes, '{color}', '"red"')
WHERE product_id = 1;

-- Add a new field
UPDATE product_jsonb
SET attributes = attributes || '{"weight": "150g"}'
WHERE product_id = 1;

-- Remove a field
UPDATE product_jsonb
SET attributes = attributes - 'weight'
WHERE product_id = 1;

JSON Fonksiyonlari ve Operatorleri:

-- Convert to/from JSON
SELECT to_json(row(1, 'test'));
SELECT to_jsonb(row(1, 'test'));

-- Build JSON objects
SELECT json_build_object('name', name, 'price', price) FROM product;

-- Aggregate to JSON
SELECT
    category_id,
    json_agg(json_build_object('id', product_id, 'name', name, 'price', price)) AS products
FROM
    product
GROUP BY
    category_id;

-- Extract keys and values
SELECT jsonb_object_keys(attributes) FROM product_jsonb;
SELECT jsonb_each(attributes) FROM product_jsonb;
SELECT jsonb_array_elements(attributes->'features') FROM product_jsonb;

JSON Yol Ifadeleri (PostgreSQL 12+):

-- Simple path expression
SELECT jsonb_path_query(
    attributes,
    '$.features[*] ? (@ == "waterproof")'
)
FROM product_jsonb;

-- Check existence
SELECT *
FROM product_jsonb
WHERE jsonb_path_exists(
    attributes,
    '$.features[*] ? (@ == "5G")'
);

Pratik JSON Ornekleri:

  1. Yapilandirma Ayarlarini Saklama
  2. Esnek Urun Ozelliklerini Yonetme
  3. Olay Verilerini Saklama

Bolum 1: Fonksiyonlar ve Sakli Yordamlar

PL/pgSQL ile yeniden kullanilabilir veritabani mantigi olusturun.

Alistirma 1.1: Temel Skaler Fonksiyon

Bir musteri icin toplam siparis degerini hesaplayan ve musteri tipine gore sadakat indirimi uygulayan bir fonksiyon olusturun.

Gereksinimler:

  • Premium musteriler %10 indirim alir
  • Normal musteriler %5 indirim alir
  • Yeni musteriler indirim almaz
CREATE OR REPLACE FUNCTION calculate_order_total(
    order_id_param INTEGER
)
RETURNS NUMERIC AS $$
DECLARE
    base_amount NUMERIC;
    customer_type_val TEXT;
    discount_rate NUMERIC;
    final_amount NUMERIC;
BEGIN
    -- Get order amount and customer type
    SELECT o.amount, c.customer_type
    INTO base_amount, customer_type_val
    FROM orders o
    JOIN customers c ON o.customer_id = c.id
    WHERE o.id = order_id_param;

    -- Check if order exists
    IF base_amount IS NULL THEN
        RAISE EXCEPTION 'Order % not found', order_id_param;
    END IF;

    -- Determine discount rate based on customer type
    CASE customer_type_val
        WHEN 'premium' THEN discount_rate := 0.10;
        WHEN 'regular' THEN discount_rate := 0.05;
        ELSE discount_rate := 0.00;
    END CASE;

    -- Calculate final amount
    final_amount := base_amount * (1 - discount_rate);

    RETURN ROUND(final_amount, 2);
END;
$$ LANGUAGE plpgsql;

-- Usage
SELECT calculate_order_total(123);
Temel Kavramlar:
  • Yerel degiskenler icin DECLARE blogu
  • Sorgu sonuclarini degiskenlere atamak icin SELECT INTO
  • Kosullu mantik icin CASE
  • Hata yonetimi icin RAISE EXCEPTION

Alistirma 1.2: Tablo Donduren Fonksiyon

Toplam harcamaya gore en iyi N musteriyi siparis istatistikleriyle birlikte donduren bir fonksiyon olusturun.

Cikti sutunlari: customer_id, customer_name, total_orders, total_spent, avg_order_value

CREATE OR REPLACE FUNCTION get_top_customers(n INTEGER)
RETURNS TABLE (
    customer_id INTEGER,
    customer_name TEXT,
    total_orders BIGINT,
    total_spent NUMERIC,
    avg_order_value NUMERIC
) AS $$
BEGIN
    RETURN QUERY
    SELECT
        c.id,
        c.name::text,
        COUNT(o.id),
        COALESCE(SUM(o.amount), 0),
        COALESCE(ROUND(AVG(o.amount), 2), 0)
    FROM customers c
    LEFT JOIN orders o ON c.id = o.customer_id
    WHERE c.deleted_at IS NULL
    GROUP BY c.id, c.name
    ORDER BY SUM(o.amount) DESC NULLS LAST
    LIMIT n;
END;
$$ LANGUAGE plpgsql;

-- Usage: Get top 10 customers
SELECT * FROM get_top_customers(10);

-- Use in a JOIN
SELECT
    tc.*,
    c.email
FROM get_top_customers(5) tc
JOIN customers c ON tc.customer_id = c.id;
RETURN QUERY ve RETURN NEXT karsilastirmasi:
  • RETURN QUERY - Bir sorgudan tum satirlari tek seferde dondurur
  • RETURN NEXT - Her seferinde bir satir dondurur (dongulerde kullanisli)

Alistirma 1.3: Hata Yonetimli Fonksiyon

Depolar arasinda envanter transferi yapan ve uygun hata yonetimi iceren bir fonksiyon olusturun.

Gereksinimler:

  • Kaynak deponun yeterli stoga sahip oldugunu kontrol edin
  • Her iki deponun da var oldugunu kontrol edin
  • Basari/basarisizlik mesaji dondur
CREATE OR REPLACE FUNCTION transfer_inventory(
    p_product_id INTEGER,
    p_source_warehouse INTEGER,
    p_dest_warehouse INTEGER,
    p_quantity INTEGER
)
RETURNS TEXT AS $$
DECLARE
    v_source_stock INTEGER;
    v_product_name TEXT;
    insufficient_stock EXCEPTION;
    warehouse_not_found EXCEPTION;
BEGIN
    -- Validate quantity
    IF p_quantity <= 0 THEN
        RAISE EXCEPTION 'Quantity must be positive';
    END IF;

    -- Check source warehouse stock
    SELECT stock_quantity, p.name
    INTO v_source_stock, v_product_name
    FROM warehouse_inventory wi
    JOIN products p ON wi.product_id = p.product_id
    WHERE wi.product_id = p_product_id
      AND wi.warehouse_id = p_source_warehouse
    FOR UPDATE;

    IF v_source_stock IS NULL THEN
        RAISE warehouse_not_found;
    END IF;

    IF v_source_stock < p_quantity THEN
        RAISE insufficient_stock;
    END IF;

    -- Deduct from source
    UPDATE warehouse_inventory
    SET stock_quantity = stock_quantity - p_quantity
    WHERE product_id = p_product_id
      AND warehouse_id = p_source_warehouse;

    -- Add to destination (upsert)
    INSERT INTO warehouse_inventory (warehouse_id, product_id, stock_quantity)
    VALUES (p_dest_warehouse, p_product_id, p_quantity)
    ON CONFLICT (warehouse_id, product_id)
    DO UPDATE SET stock_quantity = warehouse_inventory.stock_quantity + p_quantity;

    RETURN FORMAT('Successfully transferred %s units of %s', p_quantity, v_product_name);

EXCEPTION
    WHEN insufficient_stock THEN
        RETURN FORMAT('Error: Insufficient stock. Available: %s, Requested: %s',
                      v_source_stock, p_quantity);
    WHEN warehouse_not_found THEN
        RETURN 'Error: Source warehouse or product not found';
    WHEN OTHERS THEN
        RETURN FORMAT('Error: %s', SQLERRM);
END;
$$ LANGUAGE plpgsql;

-- Usage
SELECT transfer_inventory(101, 1, 2, 50);
FOR UPDATE: Bu, islem tamamlanana kadar satiri kilitler ve envanter kontrol edilirken ve guncellenirken yaris kosullarini onler.

Alistirma 1.4: Islem Kontrollu Sakli Yordam

Bir grup siparisi isleyen, basarilı olanlari onaylayan ve basarisiz olanlari geri alan bir yordam olusturun.

CREATE OR REPLACE PROCEDURE process_pending_orders()
LANGUAGE plpgsql
AS $$
DECLARE
    v_order RECORD;
    v_success_count INTEGER := 0;
    v_fail_count INTEGER := 0;
BEGIN
    -- Loop through pending orders
    FOR v_order IN
        SELECT id, customer_id, amount
        FROM orders
        WHERE status = 'pending'
        ORDER BY order_date
        LIMIT 100
    LOOP
        BEGIN
            -- Try to process this order
            UPDATE orders
            SET status = 'processing',
                updated_at = CURRENT_TIMESTAMP
            WHERE id = v_order.id;

            -- Simulate some business logic that might fail
            IF v_order.amount > 10000 THEN
                -- High-value orders need manual review
                UPDATE orders
                SET status = 'review_required'
                WHERE id = v_order.id;
            ELSE
                UPDATE orders
                SET status = 'completed'
                WHERE id = v_order.id;
            END IF;

            v_success_count := v_success_count + 1;

            -- Commit after each successful order
            COMMIT;

        EXCEPTION WHEN OTHERS THEN
            -- Rollback this order's changes
            ROLLBACK;
            v_fail_count := v_fail_count + 1;

            -- Log the error
            INSERT INTO order_processing_log (order_id, error_message, created_at)
            VALUES (v_order.id, SQLERRM, CURRENT_TIMESTAMP);
            COMMIT;
        END;
    END LOOP;

    RAISE NOTICE 'Processed: % success, % failed', v_success_count, v_fail_count;
END;
$$;

-- Usage
CALL process_pending_orders();
Yordamlar ve Fonksiyonlar:
  • Yordamlar COMMIT ve ROLLBACK kullanabilir
  • Fonksiyonlar islemleri kontrol edemez
  • Yordamlari CALL ile, fonksiyonlari SELECT ile cagirin

Bolum 2: Tetikleyiciler ve Olay Tabanli Programlama

Tetikleyiciler ile veritabani islemlerini otomatiklestirin.

Olay (INSERT/UPDATE/DELETE) → Tetikleyici Fonksiyonu → Otomatik Eylem

Alistirma 2.1: Otomatik Zaman Damgasi Guncelleme Tetikleyicisi

Bir urun degistirildiginde otomatik olarak updated_at sutununu guncelleyen bir tetikleyici olusturun.

-- Create the trigger function
CREATE OR REPLACE FUNCTION update_modified_timestamp()
RETURNS TRIGGER AS $$
BEGIN
    -- Set updated_at to current timestamp
    NEW.updated_at := CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Apply to products table
CREATE TRIGGER products_update_timestamp
BEFORE UPDATE ON products
FOR EACH ROW
EXECUTE FUNCTION update_modified_timestamp();

-- Test it
UPDATE products SET price = price * 1.1 WHERE product_id = 1;
SELECT product_id, name, updated_at FROM products WHERE product_id = 1;
Tetikleyici Zamanlama:
  • BEFORE - Yazilmadan once NEW degerlerini degistirebilir
  • AFTER - Degisiklikten sonra calisir, NEW'i degistiremez
  • INSTEAD OF - Islemi degistirir (gorunumler icin)

Alistirma 2.2: Denetim Izi Tetikleyicisi

Siparisler tablosundaki tum degisiklikleri kaydeden kapsamli bir denetim tetikleyicisi olusturun.

-- Create audit log table
CREATE TABLE orders_audit (
    audit_id SERIAL PRIMARY KEY,
    order_id INTEGER NOT NULL,
    operation VARCHAR(10) NOT NULL,
    changed_by TEXT NOT NULL DEFAULT current_user,
    changed_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    old_data JSONB,
    new_data JSONB,
    changed_fields TEXT[]
);

-- Create audit trigger function
CREATE OR REPLACE FUNCTION audit_orders_changes()
RETURNS TRIGGER AS $$
DECLARE
    v_changed_fields TEXT[] := ARRAY[]::text[];
    v_old_json JSONB;
    v_new_json JSONB;
BEGIN
    IF (TG_OP = 'DELETE') THEN
        v_old_json := to_jsonb(OLD);
        INSERT INTO orders_audit (order_id, operation, old_data)
        VALUES (OLD.id, 'DELETE', v_old_json);
        RETURN OLD;

    ELSIF (TG_OP = 'UPDATE') THEN
        v_old_json := to_jsonb(OLD);
        v_new_json := to_jsonb(NEW);

        -- Track which fields changed
        IF OLD.amount IS DISTINCT FROM NEW.amount THEN
            v_changed_fields := array_append(v_changed_fields, 'amount');
        END IF;
        IF OLD.status IS DISTINCT FROM NEW.status THEN
            v_changed_fields := array_append(v_changed_fields, 'status');
        END IF;

        INSERT INTO orders_audit (order_id, operation, old_data, new_data, changed_fields)
        VALUES (NEW.id, 'UPDATE', v_old_json, v_new_json, v_changed_fields);
        RETURN NEW;

    ELSIF (TG_OP = 'INSERT') THEN
        v_new_json := to_jsonb(NEW);
        INSERT INTO orders_audit (order_id, operation, new_data)
        VALUES (NEW.id, 'INSERT', v_new_json);
        RETURN NEW;
    END IF;

    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

-- Create the trigger
CREATE TRIGGER orders_audit_trigger
AFTER INSERT OR UPDATE OR DELETE ON orders
FOR EACH ROW
EXECUTE FUNCTION audit_orders_changes();

-- Query audit log
SELECT
    order_id,
    operation,
    changed_at,
    changed_fields,
    old_data->>'status' AS old_status,
    new_data->>'status' AS new_status
FROM orders_audit
ORDER BY changed_at DESC;

Alistirma 2.3: Is Kurali Uygulama Tetikleyicisi

Urun fiyatlandirmasi icin is kurallarini uygulayan bir tetikleyici olusturun:

  • Fiyat negatif olamaz
  • Tek bir guncellemede fiyat artisi %50'yi gecemez
  • Indirimli fiyat normal fiyattan dusuk olmalidir
CREATE OR REPLACE FUNCTION validate_product_price()
RETURNS TRIGGER AS $$
BEGIN
    -- Rule 1: Price cannot be negative
    IF NEW.price < 0 THEN
        RAISE EXCEPTION 'Product price cannot be negative (got: %)', NEW.price;
    END IF;

    -- Rule 2: Limit price increase (only for updates)
    IF TG_OP = 'UPDATE' THEN
        IF OLD.price > 0 AND NEW.price > OLD.price * 1.5 THEN
            RAISE EXCEPTION
                'Price increase exceeds 50%% limit. Old: %, New: %, Max allowed: %',
                OLD.price, NEW.price, OLD.price * 1.5;
        END IF;
    END IF;

    -- Rule 3: Discount price validation
    IF NEW.discount_price IS NOT NULL THEN
        IF NEW.discount_price >= NEW.price THEN
            RAISE EXCEPTION
                'Discount price (%) must be less than regular price (%)',
                NEW.discount_price, NEW.price;
        END IF;

        IF NEW.discount_price < 0 THEN
            RAISE EXCEPTION 'Discount price cannot be negative';
        END IF;
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER enforce_product_pricing
BEFORE INSERT OR UPDATE OF price, discount_price ON products
FOR EACH ROW
EXECUTE FUNCTION validate_product_price();

-- Test: This should fail

Alistirma 2.4: Yumusak Silme Uygulamasi

Musteriler tablosu icin yumusak silme uygulayın - kayitlari gercekten silmek yerine bir deleted_at zaman damgasi ayarlayin.

-- Trigger function for soft delete
CREATE OR REPLACE FUNCTION soft_delete_customer()
RETURNS TRIGGER AS $$
BEGIN
    -- Instead of deleting, set deleted_at
    UPDATE customers
    SET deleted_at = CURRENT_TIMESTAMP
    WHERE id = OLD.id;

    -- Return NULL to prevent the actual delete
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

-- Create trigger (only for non-deleted records)
CREATE TRIGGER customer_soft_delete
BEFORE DELETE ON customers
FOR EACH ROW
WHEN (OLD.deleted_at IS NULL)
EXECUTE FUNCTION soft_delete_customer();

-- Create a view for active customers
CREATE OR REPLACE VIEW active_customers AS
SELECT * FROM customers WHERE deleted_at IS NULL;

-- Test soft delete
DELETE FROM customers WHERE id = 1;

-- Record still exists with deleted_at set
SELECT id, name, deleted_at FROM customers WHERE id = 1;

-- But not visible in the view
SELECT * FROM active_customers WHERE id = 1; -- Returns nothing

-- Restore a soft-deleted record
UPDATE customers SET deleted_at = NULL WHERE id = 1;
Yumusak Silmenin Avantajlari:
  • Yedekleme olmadan veri kurtarma
  • Referans butunlugunu koruma
  • Silmelerin denetim izi
  • Uyumluluk gereksinimleri (veri saklama)

Bolum 3: Gelismis Indeksleme Teknikleri

Ozellestirilmis indekslerle sorgu performansini optimize edin.

Alistirma 3.1: Yaygin Sorgular icin Kismi Indeks

Sik sorgulanan veri alt kumeleri icin optimize edilmis indeksler olusturun.

Senaryo: Siparis sorgularinin %90'i 'beklemede' veya 'isleniyor' siparisleri icindir (tum siparislerin sadece %5'i).

-- Partial index for active orders only
CREATE INDEX idx_orders_active
ON orders (order_date, customer_id)
WHERE status IN ('pending', 'processing');

-- Compare with full index
CREATE INDEX idx_orders_full
ON orders (order_date, customer_id);

-- Check index sizes
SELECT
    indexrelname AS index_name,
    pg_size_pretty(pg_relation_size(indexrelid)) AS index_size
FROM pg_stat_user_indexes
WHERE relname = 'orders'
ORDER BY pg_relation_size(indexrelid) DESC;

-- Query that uses partial index
EXPLAIN ANALYZE
SELECT * FROM orders
WHERE status = 'pending'
  AND order_date > CURRENT_DATE - 7;
Indeks Tipi Indekslenen Satirlar Yaklasik Boyut
Tam Indeks 2,000,000 ~60 MB
Kismi Indeks 100,000 (%5) ~3 MB

Alistirma 3.2: Hesaplanan Degerler icin Ifade Indeksi

Fonksiyonlar ve ifadeler iceren sorgulari destekleyen indeksler olusturun.

-- Index for case-insensitive email search
CREATE INDEX idx_customers_email_lower
ON customers (LOWER(email));

-- Query MUST use the same expression
SELECT * FROM customers
WHERE LOWER(email) = 'john@example.com';

-- Index for year-month grouping
CREATE INDEX idx_orders_year_month
ON orders (
    EXTRACT(YEAR FROM order_date),
    EXTRACT(MONTH FROM order_date)
);

-- Index for JSONB attribute
CREATE INDEX idx_products_brand
ON products ((attributes->>'brand'))
WHERE attributes->>'brand' IS NOT NULL;

-- Query using JSONB index
EXPLAIN ANALYZE
SELECT * FROM products
WHERE attributes->>'brand' = 'Apple';
Ifade Eslemesi Gerekli: Sorgu indeksle AYNI ifadeyi kullanmalidir. WHERE UPPER(email) = ... ifadesi LOWER(email) uzerindeki indeksi kullanmaz.

Alistirma 3.3: Sadece Indeks Taramalari icin Kapsayici Indeks

Bir sorgunun ihtiyac duydugu tum sutunlari iceren ve tablo erisimini ortadan kaldiran bir indeks olusturun.

Sik kullanilan sorgu:

SELECT customer_id, order_date, amount, status
FROM orders
WHERE customer_id = ?
ORDER BY order_date DESC
LIMIT 10;
-- Covering index with INCLUDE
CREATE INDEX idx_orders_customer_covering
ON orders (customer_id, order_date DESC)
INCLUDE (amount, status);

-- Test the query
EXPLAIN ANALYZE
SELECT customer_id, order_date, amount, status
FROM orders
WHERE customer_id = 12345
ORDER BY order_date DESC
LIMIT 10;

-- Look for "Index Only Scan" in the plan
-- Index Only Scan using idx_orders_customer_covering
--   Index Cond: (customer_id = 12345)
--   Heap Fetches: 0  <-- No table access!
INCLUDE ve Anahtar Sutunlar:
  • Anahtar sutunlar (INCLUDE'dan once): Filtreleme ve siralama icin kullanilir
  • Dahil edilen sutunlar: Sadece veri almak icin saklanir, arama icin indekslenmez
  • Sadece SELECT'te olan, WHERE/ORDER BY'da olmayan sutunlar icin INCLUDE kullanin

Alistirma 3.4: Karmasik Analizler icin Maddilesmis Gorunum

Sik calistirilan satis ozet raporu icin maddilesmis bir gorunum olusturun.

-- Create materialized view
CREATE MATERIALIZED VIEW mv_daily_sales_summary AS
SELECT
    DATE_TRUNC('day', o.order_date) AS sale_date,
    c.category_name,
    COUNT(DISTINCT o.id) AS order_count,
    COUNT(DISTINCT o.customer_id) AS unique_customers,
    SUM(oi.quantity) AS units_sold,
    SUM(oi.quantity * oi.price) AS revenue,
    ROUND(AVG(oi.quantity * oi.price), 2) AS avg_order_value
FROM orders o
JOIN order_items oi ON o.id = oi.order_id
JOIN products p ON oi.product_id = p.product_id
JOIN categories c ON p.category_id = c.category_id
WHERE o.status = 'completed'
GROUP BY
    DATE_TRUNC('day', o.order_date),
    c.category_name
WITH DATA;

-- Create indexes on the materialized view
CREATE INDEX idx_mv_daily_sales_date
ON mv_daily_sales_summary (sale_date);

CREATE INDEX idx_mv_daily_sales_category
ON mv_daily_sales_summary (category_name);

-- Query is now super fast
SELECT
    category_name,
    SUM(revenue) AS total_revenue
FROM mv_daily_sales_summary
WHERE sale_date >= CURRENT_DATE - 30
GROUP BY category_name
ORDER BY total_revenue DESC;

-- Refresh the view (run periodically)
REFRESH MATERIALIZED VIEW mv_daily_sales_summary;

-- Concurrent refresh (doesn't block reads)
CREATE UNIQUE INDEX idx_mv_daily_sales_pk
ON mv_daily_sales_summary (sale_date, category_name);

REFRESH MATERIALIZED VIEW CONCURRENTLY mv_daily_sales_summary;
Yenileme Stratejisi:
  • Dusuk trafik donemlerinde yenileme zamanlayin
  • Okumalarin engellenmemesi gerekiyorsa CONCURRENTLY kullanin (benzersiz indeks gerektirir)
  • Cok buyuk veri setleri icin artimli yenileme kaliplarini degerlendirin

Bolum 4: Meydan Okuma Alistirmasi

Meydan Okuma: Eksiksiz Siparis Isleme Sistemi

Fonksiyonlari, tetikleyicileri ve uygun indekslemeyi birlestiren eksiksiz bir siparis isleme sistemi olusturun.

Gereksinimler:

  1. Birden fazla kalemle siparis olusturan fonksiyon
  2. Stok uygunlugunu dogrulayan tetikleyici
  3. Siparis ozet tablosunu guncelleyen tetikleyici
  4. Yaygin sorgular icin uygun indeksler
-- 1. Create supporting tables
CREATE TABLE order_summary (
    customer_id INTEGER PRIMARY KEY,
    total_orders INTEGER DEFAULT 0,
    total_spent NUMERIC DEFAULT 0,
    last_order_date DATE,
    avg_order_value NUMERIC
);

-- 2. Stock validation trigger
CREATE OR REPLACE FUNCTION check_stock_before_order_item()
RETURNS TRIGGER AS $$
DECLARE
    v_stock INTEGER;
    v_product_name TEXT;
BEGIN
    SELECT stock_level, name INTO v_stock, v_product_name
    FROM products
    WHERE product_id = NEW.product_id
    FOR UPDATE;

    IF v_stock < NEW.quantity THEN
        RAISE EXCEPTION
            'Insufficient stock for %. Available: %, Requested: %',
            v_product_name, v_stock, NEW.quantity;
    END IF;

    -- Reduce stock
    UPDATE products
    SET stock_level = stock_level - NEW.quantity
    WHERE product_id = NEW.product_id;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER validate_stock
BEFORE INSERT ON order_items
FOR EACH ROW
EXECUTE FUNCTION check_stock_before_order_item();

-- 3. Order summary update trigger
CREATE OR REPLACE FUNCTION update_order_summary()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO order_summary (customer_id, total_orders, total_spent, last_order_date, avg_order_value)
    VALUES (NEW.customer_id, 1, NEW.amount, NEW.order_date, NEW.amount)
    ON CONFLICT (customer_id)
    DO UPDATE SET
        total_orders = order_summary.total_orders + 1,
        total_spent = order_summary.total_spent + NEW.amount,
        last_order_date = GREATEST(order_summary.last_order_date, NEW.order_date),
        avg_order_value = (order_summary.total_spent + NEW.amount) / (order_summary.total_orders + 1);

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER maintain_order_summary
AFTER INSERT ON orders
FOR EACH ROW
EXECUTE FUNCTION update_order_summary();

-- 4. Main order creation function
CREATE OR REPLACE FUNCTION create_order(
    p_customer_id INTEGER,
    p_items JSONB  -- [{product_id: 1, quantity: 2}, ...]
)
RETURNS INTEGER AS $$
DECLARE
    v_order_id INTEGER;
    v_item JSONB;
    v_total NUMERIC := 0;
    v_price NUMERIC;
BEGIN
    -- Create order
    INSERT INTO orders (customer_id, order_date, status, amount)
    VALUES (p_customer_id, CURRENT_DATE, 'pending', 0)
    RETURNING id INTO v_order_id;

    -- Add items
    FOR v_item IN SELECT * FROM jsonb_array_elements(p_items)
    LOOP
        SELECT price INTO v_price
        FROM products
        WHERE product_id = (v_item->>'product_id')::integer;

        INSERT INTO order_items (order_id, product_id, quantity, price)
        VALUES (
            v_order_id,
            (v_item->>'product_id')::integer,
            (v_item->>'quantity')::integer,
            v_price
        );

        v_total := v_total + (v_price * (v_item->>'quantity')::integer);
    END LOOP;

    -- Update order total
    UPDATE orders SET amount = v_total WHERE id = v_order_id;

    RETURN v_order_id;
END;
$$ LANGUAGE plpgsql;

-- 5. Create optimized indexes
CREATE INDEX idx_orders_customer_date
ON orders (customer_id, order_date DESC)
INCLUDE (amount, status);

CREATE INDEX idx_orders_pending

  • Islem kontrollu toplu islemler icin kullanin

  • BEFORE tetikleyicileri NEW degerlerini degistirebilir
  • Gereksiz calistirmayi onlemek icin WHEN kosulu kullanin
  • Islemi iptal etmek icin BEFORE tetikleyicisinden NULL dondurun

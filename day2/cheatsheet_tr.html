Gun 2: Veri Manipulasyonu & JSON

INSERT, UPDATE, DELETE, Veri Tipleri & JSON Islemleri

DML Hizli Referans
INSERT - Yeni satirlar ekle
UPDATE - Mevcut satirlari degistir
DELETE - Satirlari sil
UPSERT - Ekle veya guncelle

INSERT Ifadesi

Tek Satir Ekleme
-- Sutunlari belirt (onerilen)
INSERT INTO employees
    (first_name, last_name, email, salary)
VALUES
    ('John', 'Doe', 'john@email.com', 50000);

-- Tum sutunlar (sira onemli)
INSERT INTO employees
VALUES (1, 'John', 'Doe', 'john@email.com', 50000);
Coklu Satir Ekleme
INSERT INTO employees
    (first_name, last_name, salary)
VALUES
    ('John', 'Doe', 50000),
    ('Jane', 'Smith', 55000),
    ('Bob', 'Johnson', 48000);

-- Eklenen satirlari dondur (PostgreSQL)
INSERT INTO employees (name, salary)
VALUES ('Alice', 60000)
RETURNING id, name;
SELECT'ten Ekleme
-- Baska tablodan veri kopyala
INSERT INTO employees_archive (id, name, salary, archived_at)
SELECT id, name, salary, NOW()
FROM employees
WHERE termination_date IS NOT NULL;

UPDATE Ifadesi

Temel UPDATE
-- Tek sutun guncelle
UPDATE employees
SET salary = 55000
WHERE id = 1;

-- Coklu sutun guncelle
UPDATE employees
SET salary = 60000,
    department_id = 5,
    updated_at = NOW()
WHERE id = 1;
Kosullu UPDATE
-- Hesaplama ile guncelle
UPDATE employees
SET salary = salary * 1.10  -- %10 zam
WHERE department_id = 5;

-- CASE ile guncelle
UPDATE employees
SET bonus = CASE
    WHEN rating >= 5 THEN salary * 0.20
    WHEN rating >= 3 THEN salary * 0.10
    ELSE salary * 0.05
END;
JOIN ile UPDATE (PostgreSQL)
UPDATE employees e
SET department_name = d.name
FROM departments d
WHERE e.department_id = d.id;
UPDATE ile her zaman WHERE kullanin! Olmadan TUM satirlar degistirilir.

DELETE Ifadesi

Temel DELETE
-- Belirli satirlari sil
DELETE FROM employees
WHERE id = 1;

-- Kosul ile sil
DELETE FROM employees
WHERE termination_date < '2023-01-01';

-- Tum satirlari sil (dikkatli!)
DELETE FROM temp_table;
Alt Sorgu ile DELETE
-- Alt sorgu kullanarak sil
DELETE FROM employees
WHERE department_id IN (
    SELECT id FROM departments
    WHERE status = 'closed'
);

-- TRUNCATE (daha hizli, WHERE yok)
TRUNCATE TABLE temp_logs;
-- Geri alinamaz, sequence'lari sifirlar

UPSERT (Ekle veya Guncelle)

ON CONFLICT (PostgreSQL)
-- Cakisma durumunda ekle veya guncelle
INSERT INTO products (sku, name, price, stock)
VALUES ('SKU001', 'Widget', 29.99, 100)
ON CONFLICT (sku)
DO UPDATE SET
    price = EXCLUDED.price,
    stock = products.stock + EXCLUDED.stock,
    updated_at = NOW();

-- Cakisma durumunda yoksay
INSERT INTO products (sku, name)
VALUES ('SKU001', 'Widget')
ON CONFLICT DO NOTHING;

JSON Operatorleri (PostgreSQL)
-- Ornek veri
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    data JSONB
);

INSERT INTO users (data) VALUES
('{"name": "John", "age": 30, "address": {"city": "NYC", "zip": "10001"}, "tags": ["admin", "active"]}');

-- JSON alanina eris (JSON dondurur)
SELECT data->'name' FROM users;              -- "John" (JSON)

-- JSON alanina metin olarak eris
SELECT data->>'name' FROM users;             -- John (text)

-- Ic ice alana eris
SELECT data->'address'->'city' FROM users;   -- "NYC"
SELECT data->'address'->>'city' FROM users;  -- NYC

-- Dizi elemanlarina eris
SELECT data->'tags'->0 FROM users;          -- "admin"

-- Yol erisimi
SELECT data#>'{address,city}' FROM users;   -- "NYC"
SELECT data#>>'{address,city}' FROM users;  -- NYC
JSON Sorgu Operatorleri
-- Icerir (sadece JSONB)
SELECT * FROM users
WHERE data @> '{"age": 30}';

-- Icerilir
WHERE '{"name": "John"}' <@ data;

-- Anahtar varligini kontrol et
WHERE data ? 'name';

-- Herhangi bir anahtar var mi
WHERE data ?| array['name', 'email'];

-- Tum anahtarlar var mi
WHERE data ?& array['name', 'age'];
JSON Fonksiyonlari
-- Tum anahtarlari al
SELECT jsonb_object_keys(data) FROM users;

-- Dizi uzunlugunu al
SELECT jsonb_array_length(data->'tags');

-- Diziyi satirlara genislet
SELECT jsonb_array_elements(data->'tags');

-- Guzel yazdir
SELECT jsonb_pretty(data) FROM users;

-- JSON nesnesi olustur
SELECT jsonb_build_object(
    'name', first_name,
    'salary', salary
) FROM employees;
JSON Degistirme (JSONB)
-- Anahtar ayarla/guncelle
UPDATE users
SET data = jsonb_set(data, '{age}', '31')
WHERE id = 1;

-- Ic ice anahtar ayarla
UPDATE users
SET data = jsonb_set(data, '{address,zip}', '"10002"');

-- Anahtar kaldir
UPDATE users
SET data = data - 'age';

-- Ic ice anahtar kaldir
UPDATE users
SET data = data #- '{address,zip}';

-- JSON birlestir
UPDATE users
SET data = data || '{"email": "john@email.com"}';
JSON Indeksleme (JSONB)
-- @>, ?, ?|, ?& operatorleri icin GIN index
CREATE INDEX idx_users_data ON users USING GIN (data);

-- Belirli yolu indeksle
CREATE INDEX idx_users_city ON users USING GIN ((data->'address'));

-- Belirli anahtar icin ifade indeksi
CREATE INDEX idx_users_name ON users ((data->>'name'));

Transaction'lar

Transaction Kontrolu
-- Transaction baslat
BEGIN;  -- veya START TRANSACTION

-- DML ifadeleriniz
UPDATE accounts SET balance = balance - 100 WHERE id = 1;
UPDATE accounts SET balance = balance + 100 WHERE id = 2;

-- Degisiklikleri kaydet
COMMIT;

-- Veya degisiklikleri geri al
ROLLBACK;

-- Kismi geri alma icin savepoint
BEGIN;
INSERT INTO orders (...) VALUES (...);
SAVEPOINT before_items;
INSERT INTO order_items (...) VALUES (...);
ROLLBACK TO before_items;  -- Sadece item'lari geri al
COMMIT;  -- Siparis kaydedilir

En Iyi Uygulamalar

UPDATE/DELETE ile her zaman WHERE kullanin - Kazara toplu degisiklikleri onleyin
Transaction kullanin - Iliskili degisiklikleri birlikte gruplayin
Para icin DECIMAL kullanin - Finansal veri icin asla FLOAT kullanmayin
TIMESTAMPTZ kullanin - Zaman damgalarini her zaman saat dilimi ile saklayin
JSON yerine JSONB - Daha iyi performans ve indeksleme
DML'i transaction'da test edin - BEGIN, ifadeyi calistir, dogrula, sonra COMMIT veya ROLLBACK

Gun 1: SQL Temelleri

Veritabanlari, SELECT, JOIN'ler & Gelismis Sorgular

Hizli Referans - SQL Sorgu Sirasi
SELECT sutunlar      -- 5. Sutunlari sec
FROM tablo           -- 1. Tabloyu sec
JOIN tablo2 ON ...  -- 2. Tablolari birlestir
WHERE kosul          -- 3. Satirlari filtrele
GROUP BY sutun       -- 4. Satirlari grupla
HAVING kosul         -- 6. Gruplari filtrele
ORDER BY sutun       -- 7. Sonuclari sirala
LIMIT n              -- 8. Satir sayisini sinirla

Veritabani Temelleri

Temel Kavramlar
  • Database: Yapilandirilmis verilerin organize koleksiyonu
  • Table: Satir ve sutunlarda iliskili veri koleksiyonu
  • Row (Record): Tablodaki tek bir kayit
  • Column (Field): Verinin bir ozelligi
  • Schema: Veritabani yapisinin semasi
Anahtarlar & Kisitlamalar
  • PRIMARY KEY: Her satir icin benzersiz tanimlayici
  • FOREIGN KEY: Baska tablonun primary key'ine baglanti
  • UNIQUE: Tekrar eden degerlere izin verilmez
  • NOT NULL: Deger mutlaka saglanmali
  • CHECK: Deger belirli kosulu karsilamali

SELECT Ifadesi

Temel SELECT Sozdizimi
-- Tum sutunlari sec
SELECT * FROM employees;

-- Belirli sutunlari sec
SELECT first_name, last_name, salary
FROM employees;

-- Sutun takma adi
SELECT first_name AS "Ad",
       salary * 12 AS yillik_maas
FROM employees;
-- Tekrarlari kaldir
SELECT DISTINCT department_id
FROM employees;

-- Sonuclari sinirla
SELECT * FROM employees
LIMIT 10;

-- Satirlari atla (sayfalama)
SELECT * FROM employees
LIMIT 10 OFFSET 20;

WHERE ile Filtreleme

Karsilastirma Operatorleri
SELECT * FROM employees WHERE
  salary = 50000          -- Esit
  salary != 50000         -- Esit degil
  salary > 50000          -- Buyuk
  salary >= 50000         -- Buyuk veya esit
  salary < 50000          -- Kucuk
  salary <= 50000         -- Kucuk veya esit
  salary BETWEEN 40000 AND 60000
  department_id IN (1, 2, 3)
  first_name LIKE 'J%'    -- J ile baslar
  email IS NULL           -- NULL kontrolu
  email IS NOT NULL
Mantiksal Operatorler
-- AND: Her iki kosul da dogru olmali
SELECT * FROM employees
WHERE salary > 50000
  AND department_id = 5;

-- OR: En az bir kosul dogru olmali
SELECT * FROM employees
WHERE department_id = 1
  OR department_id = 2;

-- NOT: Kosulu olumsuzla
SELECT * FROM employees
WHERE NOT department_id = 5;

-- Parantezlerle birlesik
WHERE (dept = 1 OR dept = 2)
  AND salary > 50000

Toplama Fonksiyonlari

Yaygin Toplama Fonksiyonlari
COUNT(*)        -- Tum satirlari say
COUNT(sutun)    -- NULL olmayan degerleri say
SUM(sutun)      -- Degerlerin toplami
AVG(sutun)      -- Ortalama deger
MIN(sutun)      -- Minimum deger
MAX(sutun)      -- Maksimum deger
GROUP BY & HAVING
SELECT department_id,
       COUNT(*) AS calisan_sayisi,
       AVG(salary) AS ort_maas
FROM employees
GROUP BY department_id
HAVING COUNT(*) > 5
ORDER BY ort_maas DESC;

Alt Sorgular

Skaler Alt Sorgu (Tek Deger)
-- Ortalamadan fazla kazanan calisanlar
SELECT * FROM employees
WHERE salary > (
    SELECT AVG(salary)
    FROM employees
);
Liste Alt Sorgusu (Coklu Deger)
-- Satis veya Pazarlamadaki calisanlar
SELECT * FROM employees
WHERE department_id IN (
    SELECT id FROM departments
    WHERE name IN ('Sales', 'Marketing')
);
EXISTS Alt Sorgusu
-- Calisani olan departmanlar
SELECT * FROM departments d
WHERE EXISTS (
    SELECT 1 FROM employees e
    WHERE e.department_id = d.id
);

Ortak Tablo Ifadeleri (CTE)

WITH Cumleciegi
WITH dept_stats AS (
    SELECT department_id,
           AVG(salary) AS ort_maas,
           COUNT(*) AS calisan_sayisi
    FROM employees
    GROUP BY department_id
)
SELECT d.name, ds.ort_maas, ds.calisan_sayisi
FROM departments d
JOIN dept_stats ds ON d.id = ds.department_id
WHERE ds.ort_maas > 50000;
CTE'ler okunabilirligi arttirir ve ana sorguda birden fazla kez referans edilebilir.

Pencere Fonksiyonlari

Pencere Fonksiyonu Sozdizimi
SELECT
    employee_name,
    department_id,
    salary,
    ROW_NUMBER() OVER (ORDER BY salary DESC) AS siralama,
    RANK() OVER (PARTITION BY department_id ORDER BY salary DESC) AS dept_sirasi,
    SUM(salary) OVER (PARTITION BY department_id) AS dept_toplam,
    AVG(salary) OVER () AS sirket_ort
FROM employees;

Sonuclari Siralama

ORDER BY Ornekleri
-- Artan (varsayilan)
SELECT * FROM employees ORDER BY last_name;

-- Azalan
SELECT * FROM employees ORDER BY salary DESC;

-- Coklu sutun
SELECT * FROM employees
ORDER BY department_id ASC, salary DESC;

-- NULL'lari yonetme
ORDER BY commission NULLS LAST;
ORDER BY commission NULLS FIRST;

En Iyi Uygulamalar

Anlamli takma adlar kullanin - Karmasik sorgulari okunabilir kilar
Sutunlari belirtin - Uretimde SELECT * kullanmayin
Tablo takma adlari kullanin - Self-join'lerde gerekli, genel olarak iyi aliskanlik
NULL, NULL'a esit degildir - = NULL degil, IS NULL kullanin
GROUP BY tuzagi - Toplanmamis tum sutunlar GROUP BY'da olmali
ON olmadan JOIN - Kartezyen carpim olusturur (genellikle hatali)
